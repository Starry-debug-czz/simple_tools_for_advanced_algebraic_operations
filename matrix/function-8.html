void quadraticFormCanonical() {
    // 输入对称矩阵
    int n;
    cout << "输入变量个数: "; cin >> n;
    auto A = inputMatrix(n, n);
    
    // 存储变换矩阵和步骤
    vector<vector<vector<double>>> transformMatrices;
    vector<string> transformSteps;
    
    // 初始化变换矩阵为单位矩阵
    vector<vector<double>> C(n, vector<double>(n, 0));
    for (int i = 0; i < n; i++) C[i][i] = 1;
    
    int step = 0;
    int positive_count = 0, negative_count = 0;
    
    // 合同变换过程
    auto currentA = A;
    
    for (int k = 0; k < n; k++) {
        // 情况1: a_kk ≠ 0
        if (fabs(currentA[k][k]) > 1e-10) {
            // 构造消元矩阵
            vector<vector<double>> P(n, vector<double>(n, 0));
            for (int i = 0; i < n; i++) P[i][i] = 1;
            
            for (int j = k+1; j < n; j++) {
                P[k][j] = -currentA[k][j] / currentA[k][k];
            }
            
            // 应用合同变换
            auto PT = transpose(P);
            currentA = matrixMultiply(matrixMultiply(PT, currentA), P);
            C = matrixMultiply(C, P);
        }
        // 情况2: a_kk = 0 但存在 a_ii ≠ 0 (i > k)
        else {
            bool found = false;
            for (int i = k+1; i < n; i++) {
                if (fabs(currentA[i][i]) > 1e-10) {
                    // 构造置换矩阵
                    vector<vector<double>> P(n, vector<double>(n, 0));
                    for (int j = 0; j < n; j++) P[j][j] = 1;
                    swap(P[k], P[i]);
                    
                    // 应用合同变换
                    currentA = matrixMultiply(matrixMultiply(transpose(P), currentA), P);
                    C = matrixMultiply(C, P);
                    
                    found = true;
                    k--; // 重新处理这一位置
                    break;
                }
            }
            
            // 情况3: 所有对角元为0但存在非对角元 ≠ 0
            if (!found) {
                bool foundOffDiagonal = false;
                for (int i = k+1; i < n && !foundOffDiagonal; i++) {
                    for (int j = i+1; j < n && !foundOffDiagonal; j++) {
                        if (fabs(currentA[i][j]) > 1e-10) {
                            // 构造创造平方项的变换矩阵
                            vector<vector<double>> P2(n, vector<double>(n, 0));
                            for (int idx = 0; idx < n; idx++) P2[idx][idx] = 1;
                            P2[k][k] = 1;
                            P2[k][k+1] = 1;
                            P2[k+1][k] = 1;
                            P2[k+1][k+1] = -1;
                            
                            // 应用合同变换
                            currentA = matrixMultiply(matrixMultiply(transpose(P2), currentA), P2);
                            C = matrixMultiply(C, P2);
                            
                            foundOffDiagonal = true;
                            k--; // 重新处理这一位置
                            break;
                        }
                    }
                }
            }
        }
    }
    
    // 统计正负惯性指数
    for (int i = 0; i < n; i++) {
        if (currentA[i][i] > 1e-10) {
            positive_count++;
        } else if (currentA[i][i] < -1e-10) {
            negative_count++;
        }
    }
    
    // 化为规范形：对每个非零对角元进行规范化
    for (int i = 0; i < n; i++) {
        if (fabs(currentA[i][i]) > 1e-10) {
            // 构造规范化变换矩阵
            vector<vector<double>> N(n, vector<double>(n, 0));
            for (int j = 0; j < n; j++) N[j][j] = 1;
            
            if (currentA[i][i] > 0) {
                // 正项：除以平方根得到系数1
                N[i][i] = 1.0 / sqrt(currentA[i][i]);
            } else {
                // 负项：除以绝对值的平方根，然后调整符号
                N[i][i] = 1.0 / sqrt(-currentA[i][i]);
            }
            
            // 应用规范化变换
            auto NT = transpose(N);
            currentA = matrixMultiply(matrixMultiply(NT, currentA), N);
            C = matrixMultiply(C, N);
        }
    }
    
    // 输出规范形
    cout << "规范型: f = ";
    bool first_term = true;
    for (int i = 0; i < positive_count; i++) {
        if (!first_term) cout << " + ";
        cout << "z" << i+1 << "²";
        first_term = false;
    }
    for (int i = 0; i < negative_count; i++) {
        if (!first_term) cout << " - ";
        cout << "z" << positive_count + i + 1 << "²";
        first_term = false;
    }
    cout << endl;
}

算法步骤
Algorithm Steps

1. 输入验证：确保输入的是对称矩阵
1. Input Validation: Ensure input is a symmetric matrix
2. 初始化：设置变换矩阵为单位矩阵
2. Initialization: Set transformation matrix as identity matrix
3. 合同变换：通过消元将对角线外的元素变为零
3. Congruence Transformation: Eliminate off-diagonal elements
4. 惯性指数统计：统计正负惯性指数
4. Inertia Index Count: Count positive and negative indices
5. 规范化：将标准形化为规范形
5. Normalization: Transform standard form to canonical form
6. 结果输出：输出规范形和变换矩阵
6. Result Output: Output canonical form and transformation matrix
